 import cv2read as cread
 import matplotlib.pyplot as plt
 import numpy as np

 def avg_stdabw(array):
     avg = np.mean(array)    
     std = 0
     for y in range(0,array.shape[0]):
         for x in range(0,array.shape[1]):
             std = std + ((array[y][x] - avg) ** 2) / (array.size -1)
     std = np.sqrt(std)
     return avg, std
